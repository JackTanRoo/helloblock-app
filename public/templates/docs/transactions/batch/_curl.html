DEFINITION
  
GET https://api.helloblock.io/v1/transactions/?tx_hashes[]=<address>
EXAMPLE REQUEST
  
# Note: option -g enables globbing so [] can be parsed

$ curl -G https://api.helloblock.io/v1/transactions \
   -d "tx_hashes[]=mu1izpJmF7CHnbVcH59f1PqfvXnmiBEMq8" \
   -d "tx_hashes[]=mvANEVQRsAC7xHt4GW2iSsJwUSMgkGX2k3" \
   -d "tx_hashes[]=n1Fr4r3wEcbXZGQJaqJkFRiMpYLAsk81RX" \
   -d "tx=true" \
   -d "wallet=true" \
   -d "info=true"
